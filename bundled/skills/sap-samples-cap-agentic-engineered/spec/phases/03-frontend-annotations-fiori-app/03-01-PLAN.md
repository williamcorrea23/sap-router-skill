4. **Create `test/integration/annotations.test.js`** (covers UI-01, UI-02, UI-03):
This test uses `cds.compile` to load the full model and verify annotations exist.

**IMPORTANT (robustness for 03-02 compatibility):** The high-importance count assertion MUST count only `DataField` entries (items with a `Value` property and `@UI.Importance: #High`), NOT `DataFieldForAction` entries. Plan 03-02 will remove the `DataFieldForAction` from LineItem and replace it with a custom action in manifest.json. If the test counts `DataFieldForAction` entries, it will break when 03-02 runs. The 8 GL data fields + 3 risk data fields = 11 `DataField` entries with `#High` importance, which is stable regardless of whether `DataFieldForAction` is present.

```javascript
const cds = require('@sap/cds');
const path = require('path');

describe('CDS UI Annotations', () => {
  let csn;

  beforeAll(async () => {
    // Compile the full CDS model from project root
    csn = await cds.load(path.join(__dirname, '../../'));
  });

  test('GLTransactions entity has UI.LineItem annotation', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    expect(entity).toBeDefined();
    expect(entity['@UI.LineItem']).toBeDefined();
    expect(Array.isArray(entity['@UI.LineItem'])).toBe(true);
  });

  test('UI.LineItem has at least 11 high-importance DataField entries (8 GL + 3 risk)', () => {
    const lineItem = csn.definitions['RiskService.GLTransactions']['@UI.LineItem'];
    // Count ONLY DataField entries with @UI.Importance #High (NOT DataFieldForAction).
    // DataField entries have a Value property; DataFieldForAction has an Action property instead.
    // This assertion is stable even after 03-02 removes DataFieldForAction from LineItem.
    const highImportanceDataFields = lineItem.filter(
      item => item.Value && item['@UI.Importance'] === '#High'
    );
    expect(highImportanceDataFields.length).toBeGreaterThanOrEqual(11);
  });

  test('UI.SelectionFields includes riskClassification and CompanyCode', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    const selFields = entity['@UI.SelectionFields'];
    expect(selFields).toBeDefined();
    expect(Array.isArray(selFields)).toBe(true);
    // SelectionFields are path expressions
    const fieldNames = selFields.map(f => f.$Path || f);
    expect(fieldNames).toContain('riskClassification');
    expect(fieldNames).toContain('CompanyCode');
  });

  test('entity has UI.Criticality pointing to criticality field (UI-03)', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    // @UI.Criticality can be a path expression
    const crit = entity['@UI.Criticality'];
    expect(crit).toBeDefined();
  });

  test('entity has virtual criticality Integer field', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    const critField = entity.elements.criticality;
    expect(critField).toBeDefined();
    expect(critField.virtual).toBe(true);
  });

  test('UI.PresentationVariant has PostingDate descending sort', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    const pv = entity['@UI.PresentationVariant'];
    expect(pv).toBeDefined();
    expect(pv.SortOrder).toBeDefined();
    expect(pv.SortOrder[0].Property).toBe('PostingDate');
    expect(pv.SortOrder[0].Descending).toBe(true);
  });

  test('UI.HeaderInfo has correct entity type names', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    const hi = entity['@UI.HeaderInfo'];
    expect(hi).toBeDefined();
    expect(hi.TypeName).toMatch(/i18n>Transaction/);
    expect(hi.TypeNamePlural).toMatch(/i18n>Transactions/);
  });

  test('UI.DataPoint for anomalyScore exists', () => {
    const entity = csn.definitions['RiskService.GLTransactions'];
    const dp = entity['@UI.DataPoint#anomalyScore'];
    expect(dp).toBeDefined();
    expect(dp.Value).toBeDefined();
  });
});
```
