```python
"""
Pydantic models for SM37 (Job Log Monitoring) lookup tool.

These models represent background job metadata and job logs retrieved from SM37.
"""

from pydantic import AwareDatetime, BaseModel, Field

from sapguimcp.models.base import ToolResult

__all__ = [
    "SM37Job",
    "SM37JobLog",
    "SM37JobListResult",
]


class SM37Job(BaseModel):
    """A background job entry from SM37 job list."""

    job_name: str = Field(description="Job name")
    job_number: str = Field(description="Job number (unique within job name)")
    status: str = Field(description="Job status: Scheduled/Released/Active/Finished/Canceled")
    start_time: str | None = Field(default=None, description="Job start date/time")
    end_time: str | None = Field(default=None, description="Job end date/time")
    duration: str | None = Field(default=None, description="Job duration")
    user: str = Field(description="User who scheduled the job")


class SM37JobLog(BaseModel):
    """Job log detail for a single background job."""

    job_name: str = Field(description="Job name")
    job_number: str = Field(description="Job number")
    log_lines: list[str] = Field(default_factory=list, description="Job log message lines")


class SM37JobListResult(ToolResult):
    """Result of SM37 job list lookup."""

    jobs: list[SM37Job] = Field(default_factory=list, description="List of matching jobs")
    job_count: int = Field(default=0, description="Number of jobs returned")
    filters_applied: dict[str, str] = Field(
        default_factory=dict,
        description="Summary of filters that were applied",
    )
    job_log: SM37JobLog | None = Field(
        default=None,
        description="Job log (only populated when include_log=True and exactly one job matches)",
    )
    retrieved_at: AwareDatetime = Field(description="When the data was retrieved")
```

```python
def format_sap_date(iso_date: str, language: str) -> str:
    """
    Convert ISO date (YYYY-MM-DD) to SAP locale format.

    Args:
        iso_date: Date string in YYYY-MM-DD format (e.g., "2026-02-22")
        language: SAP language code ("DE" or "EN")

    Returns:
        Formatted date string:
        - DE: DD.MM.YYYY (e.g., "22.02.2026")
        - EN: MM/DD/YYYY (e.g., "02/22/2026")

    Raises:
        ValueError: If iso_date is not in YYYY-MM-DD format
    """
    parts = iso_date.split("-")
    if len(parts) != 3:
        raise ValueError(f"Expected YYYY-MM-DD format, got: {iso_date}")

    year, month, day = parts[0], parts[1], parts[2]

    if language.upper() == "DE":
        return f"{day}.{month}.{year}"
    else:
        return f"{month}/{day}/{year}"
```

```python
async def capture_yaml_snapshot(
    client: ClientSession,
    base_name: str,
    overwrite: bool = False,
) -> str:
    """Capture YAML accessibility snapshot for parser development."""
    result = await call_tool_typed(client, "browser_snapshot", {}, SnapshotResult)
    yaml_content = result.snapshot

    language = os.environ.get("SAP_LANGUAGE", "de").lower()
    filename = f"{base_name}_{language}.yaml"
    filepath = YAML_SNAPSHOTS_DIR / filename

    if not filepath.exists() or overwrite:
        YAML_SNAPSHOTS_DIR.mkdir(parents=True, exist_ok=True)
        filepath.write_text(yaml_content, encoding="utf-8")
        print(f"Saved YAML snapshot: {filepath}")
```

### Task 6: Write the SM37 Parser

**Files:** Create `src/sapguimcp/parsers/sm37_parser.py`

**Step 1:** Create the parser file. The parser extracts job data from ARIA snapshots of the SM37 job list screen. The exact regex patterns will need to be refined after examining the YAML snapshots captured in Task 5, but the structure follows the SE37 parser pattern:

```python
"""
Parser for SM37 (Job Overview) ARIA snapshots.

Extracts background job list and job log data from SM37 display screens.
Handles both German and English SAP installations.
"""

import logging
import re
from datetime import UTC, datetime

from sapguimcp.lang import SM37_STATUS_MAP_DE, SM37_STATUS_MAP_EN
from sapguimcp.models.sm37_models import SM37Job, SM37JobLog

logger = logging.getLogger(__name__)

__all__ = [
    "parse_sm37_job_list",
    "parse_sm37_job_log",
]


# =============================================================================
# Regex Patterns
# =============================================================================

# Job list row pattern - SM37 job list is an ALV grid with columns:
# Jobname | Jobersteller | Status | Startdatum | Startzeit | Enddatum | Endzeit | Dauer | Jobnummer
# (EN: Job name | Job creator | Status | Start date | Start time | End date | End time | Duration | Job number)
#
# The row format in ARIA snapshots follows the pattern:
#   row "JOBNAME USERNAME STATUS DATE TIME DATE TIME DURATION NUMBER" :
#
# NOTE: This pattern will be refined based on actual YAML snapshot structure
# captured by the exploration tests in Task 5. The ARIA snapshot format
# depends on the specific ALV grid rendering in WebGUI.
_JOB_ROW_PATTERN = re.compile(
    r'row\s+"(?P<content>[^"]+)"\s*:',
    re.IGNORECASE,
)

# Column header pattern to detect we're on the job list screen
_COLUMN_HEADER_PATTERN = re.compile(
    r'columnheader\s+"(?P<name>[^"]+)"',
    re.IGNORECASE,
)

# Status bar messages indicating no jobs found
_NO_JOBS_PATTERNS = [
    "keine jobs gefunden",
    "no jobs found",
    "keine selektierten jobs",
    "no selected jobs",
    "kein job selektiert",
    "no job selected",
]

# Job log line pattern
_LOG_LINE_PATTERN = re.compile(
    r'(?:row|text|paragraph)\s+"(?P<line>[^"]+)"',
    re.IGNORECASE,
)


# =============================================================================
# Parser Functions
# =============================================================================


def _normalize_status(status_text: str) -> str:
    """
    Normalize a DE/EN status text to the canonical English form.

    Args:
        status_text: Status text from SM37 (e.g., "Beendet", "Finished", "Aktiv")

    Returns:
        Normalized English status string.
    """
    status_clean = status_text.strip()

    # Try DE map first
    if status_clean in SM37_STATUS_MAP_DE:
        return SM37_STATUS_MAP_DE[status_clean]

    # Try EN map
    if status_clean in SM37_STATUS_MAP_EN:
        return SM37_STATUS_MAP_EN[status_clean]

    # Return as-is if not recognized
    return status_clean


def _detect_columns(snapshot: str) -> list[str]:
    """
    Detect column headers from the job list ALV grid.

    Returns list of column names in order. Used to understand the
    structure of the grid and parse rows correctly.
    """
    columns: list[str] = []
    for match in _COLUMN_HEADER_PATTERN.finditer(snapshot):
        col_name = match.group("name").strip()
        if col_name:
            columns.append(col_name)
    return columns


def is_no_jobs_found(snapshot: str) -> bool:
    """Check if the snapshot indicates no jobs were found."""
    snapshot_lower = snapshot.lower()
    return any(pattern in snapshot_lower for pattern in _NO_JOBS_PATTERNS)


def is_job_list_screen(snapshot: str) -> bool:
    """Check if we're on the job list results screen (not the selection screen)."""
    # Job list screen has column headers like "Jobname"/"Job name" in a grid
    columns = _detect_columns(snapshot)
    job_col_names = {"Jobname", "Job name", "Job Name"}
    return bool(set(columns) & job_col_names)


def parse_sm37_job_list(snapshot: str) -> list[SM37Job]:
    """
    Parse the SM37 job list from an ARIA snapshot.

    The job list is an ALV grid. Column order and exact row format
    depend on the SAP system configuration. This parser:
    1. Detects column headers to determine column order
    2. Parses each row using the detected column positions
    3. Normalizes status values to English

    Args:
        snapshot: YAML accessibility snapshot of the SM37 job list screen

    Returns:
        List of SM37Job objects. Empty list if no jobs found or parse fails.

    NOTE: The row parsing logic below uses a column-based approach.
    After capturing real YAML snapshots (Task 5), this function
    will be updated with the exact row format. The current implementation
    provides the structural framework.
    """
    if is_no_jobs_found(snapshot):
        return []

    columns = _detect_columns(snapshot)
    if not columns:
        logger.warning("No column headers found in SM37 snapshot")
        return []

    # Build column index map (case-insensitive matching)
    col_map: dict[str, int] = {}
    for idx, col in enumerate(columns):
        col_map[col.lower()] = idx

    # Known column name variants (DE/EN)
    job_name_keys = ["jobname", "job name"]
    user_keys = ["jobersteller", "job creator", "ersteller", "creator", "benutzername", "user name"]
    status_keys = ["status"]
    start_date_keys = ["startdatum", "start date"]
    start_time_keys = ["startzeit", "start time"]
    end_date_keys = ["enddatum", "end date"]
    end_time_keys = ["endzeit", "end time"]
    duration_keys = ["dauer", "duration"]
    job_number_keys = ["jobnummer", "job number", "job count"]

    def _find_col_idx(keys: list[str]) -> int | None:
        for key in keys:
            if key in col_map:
                return col_map[key]
        return None

    job_name_idx = _find_col_idx(job_name_keys)
    user_idx = _find_col_idx(user_keys)
    status_idx = _find_col_idx(status_keys)
    start_date_idx = _find_col_idx(start_date_keys)
    start_time_idx = _find_col_idx(start_time_keys)
    end_date_idx = _find_col_idx(end_date_keys)
    end_time_idx = _find_col_idx(end_time_keys)
    duration_idx = _find_col_idx(duration_keys)
    job_number_idx = _find_col_idx(job_number_keys)

    if job_name_idx is None:
        logger.warning("Could not find job name column", extra={"columns": columns})
        return []

    # Parse rows - each row in the ALV grid is represented as cells
    # The exact parsing depends on the ARIA snapshot structure.
    # SM37 rows in SAP WebGUI ARIA snapshots typically appear as:
    #   row "cell1 cell2 cell3 ..." :
    # or as individual cell elements within a row.
    #
    # This implementation parses whitespace-separated values within row strings.
    # It will be refined after examining actual snapshots.
    jobs: list[SM37Job] = []

    for match in _JOB_ROW_PATTERN.finditer(snapshot):
        content = match.group("content").strip()

        # Skip header rows and empty rows
        if not content or content.startswith("Zum Auswählen") or content.startswith("To select"):
            continue

        # Skip if this looks like a column header row
        col_lower = content.lower()
        if any(key in col_lower for key in ["jobname", "job name", "jobersteller", "job creator"]):
            # Check if it contains MULTIPLE column header names (it's a header row)
            header_matches = sum(1 for key in columns if key.lower() in col_lower)
            if header_matches >= 3:
                continue

        # Split row content into cells
        # NOTE: This naive whitespace split will be replaced with the actual
        # cell structure observed in YAML snapshots. SAP ALV grids may use
        # individual cell elements or space-separated values.
        cells = content.split()

        if len(cells) < len(columns):
            # Row has fewer cells than columns - might be a different format
            # or a row with empty trailing cells
            logger.debug("Row has fewer cells than columns", extra={"cells": len(cells), "columns": len(columns)})
            continue

        def _get_cell(idx: int | None) -> str:
            if idx is not None and idx < len(cells):
                return cells[idx]
            return ""

        job_name = _get_cell(job_name_idx)
        user = _get_cell(user_idx)
        status_raw = _get_cell(status_idx)
        start_date = _get_cell(start_date_idx)
        start_time = _get_cell(start_time_idx)
        end_date = _get_cell(end_date_idx)
        end_time = _get_cell(end_time_idx)
        duration = _get_cell(duration_idx)
        job_number = _get_cell(job_number_idx)

        if not job_name:
            continue

        # Combine date + time
        start_combined = f"{start_date} {start_time}".strip() if start_date else None
        end_combined = f"{end_date} {end_time}".strip() if end_date else None

        jobs.append(
            SM37Job(
                job_name=job_name,
                job_number=job_number or "0",
                status=_normalize_status(status_raw),
                start_time=start_combined if start_combined else None,
                end_time=end_combined if end_combined else None,
                duration=duration if duration else None,
                user=user or "",
            )
        )

    return jobs


def parse_sm37_job_log(snapshot: str, job_name: str, job_number: str) -> SM37JobLog:
    """
    Parse the SM37 job log from an ARIA snapshot.

    The job log screen shows log messages as a list/table of text lines.

    Args:
        snapshot: YAML accessibility snapshot of the SM37 job log screen
        job_name: Name of the job (for the result object)
        job_number: Number of the job (for the result object)

    Returns:
        SM37JobLog with extracted log lines.
    """
    log_lines: list[str] = []

    for match in _LOG_LINE_PATTERN.finditer(snapshot):
        line = match.group("line").strip()
        if line and len(line) > 2:  # Skip very short noise
            # Skip navigation hints and UI chrome
            if line.startswith("Zum Auswählen") or line.startswith("To select"):
                continue
            log_lines.append(line)

    return SM37JobLog(
        job_name=job_name,
        job_number=job_number,
        log_lines=log_lines,
    )
```

**Important note:** The regex patterns in this parser are preliminary. After running the exploration tests (Task 5) and examining the actual YAML snapshot structure, the patterns should be refined. The column-based parsing approach is sound, but the exact cell extraction logic depends on how SAP WebGUI renders the ALV grid in the ARIA tree.

```python
"""
SM37 (Job Overview) lookup tool.

This module provides a read-only tool to list background jobs from SM37
with status/date/user filters, and optionally retrieve job logs.
"""

import json
import logging
import re
from datetime import UTC, datetime
from pathlib import Path
from typing import Any, Literal

from fastmcp import FastMCP
from mcp.types import ToolAnnotations

from sapguimcp.models import get_browser_manager
from sapguimcp.models.config import get_settings
from sapguimcp.models.sm37_models import SM37Job, SM37JobListResult, SM37JobLog
from sapguimcp.parsers.sm37_parser import is_no_jobs_found, parse_sm37_job_list, parse_sm37_job_log
from sapguimcp.tools.sap_tool_impl import sap_fill_form_impl, sap_press_key_impl, sap_transaction_impl
from sapguimcp.utils import format_sap_date

logger = logging.getLogger(__name__)

__all__ = ["register_sm37_tools"]

# Maximum number of jobs to return (cap for safety)
MAX_JOBS = 200

# Status checkbox mapping: internal name -> (DE label, EN label)
STATUS_CHECKBOX_MAP: dict[str, tuple[str, str]] = {
    "scheduled": ("Eingeplant", "Scheduled"),
    "released": ("Freigegeben", "Released"),
    "active": ("Aktiv", "Active"),
    "finished": ("Beendet", "Finished"),
    "canceled": ("Abgebrochen", "Canceled"),
}

# All internal status names
ALL_STATUSES = list(STATUS_CHECKBOX_MAP.keys())


# =============================================================================
# SM37 Navigation Helpers
# =============================================================================


def _get_checkbox_label(status: str, language: str) -> str:
    """Get the DE/EN checkbox label for a status."""
    de_label, en_label = STATUS_CHECKBOX_MAP[status]
    return de_label if language.upper() == "DE" else en_label


async def _set_status_checkboxes(page: Any, statuses: list[str], language: str) -> list[str]:
    """
    Set status checkboxes on the SM37 selection screen.

    If statuses is empty or contains all statuses, leave checkboxes unchanged
    (SAP default shows all). Otherwise, uncheck all and check only requested.

    Args:
        page: Playwright page object
        statuses: List of status names to check (e.g., ["finished", "canceled"])
        language: SAP language ("DE" or "EN")

    Returns:
        List of error messages (empty if all succeeded).
    """
    # If no specific filter or all statuses requested, leave checkboxes as-is
    if not statuses or set(statuses) == set(ALL_STATUSES):
        return []

    errors: list[str] = []

    for status_name in ALL_STATUSES:
        label = _get_checkbox_label(status_name, language)
        should_be_checked = status_name in statuses

        try:
            # Find checkbox by label
            checkbox = page.get_by_role("checkbox", name=re.compile(re.escape(label), re.IGNORECASE))

            if await checkbox.count() == 0:
                errors.append(f"Checkbox '{label}' not found")
                continue

            # Get current state
            is_checked = await checkbox.is_checked()

            # Toggle if needed
            if should_be_checked and not is_checked:
                await checkbox.check()
                await page.wait_for_timeout(100)
            elif not should_be_checked and is_checked:
                await checkbox.uncheck()
                await page.wait_for_timeout(100)

        except Exception as e:  # pylint: disable=broad-exception-caught
            errors.append(f"Failed to set checkbox '{label}': {e}")
            logger.warning("Checkbox error", extra={"label": label, "error": str(e)})

    return errors


async def _fill_selection_screen(
    page: Any,
    job_name: str,
    username: str | None,
    statuses: list[str] | None,
    from_date: str | None,
    to_date: str | None,
    language: str,
) -> list[str]:
    """
    Fill the SM37 selection screen fields.

    Args:
        page: Playwright page object
        job_name: Job name filter (supports wildcards)
        username: Username filter (None = leave default)
        statuses: Status filter list (None = leave all checked)
        from_date: Start date in ISO format (None = leave default)
        to_date: End date in ISO format (None = leave default)
        language: SAP language ("DE" or "EN")

    Returns:
        List of error messages (empty if all succeeded).
    """
    errors: list[str] = []

    # Fill job name field
    job_name_labels = {"Jobname": job_name} if language == "DE" else {"Job name": job_name}
    fill_result = await sap_fill_form_impl(job_name_labels, strict=False)
    if fill_result.not_found:
        # Try the other language as fallback
        alt_labels = {"Job name": job_name} if language == "DE" else {"Jobname": job_name}
        fill_result = await sap_fill_form_impl(alt_labels, strict=False)
        if fill_result.not_found:
            errors.append(f"Could not find job name field: {fill_result.not_found}")

    # Fill username field
    if username is not None:
        user_labels = {"Benutzername": username} if language == "DE" else {"User name": username}
        fill_result = await sap_fill_form_impl(user_labels, strict=False)
        if fill_result.not_found:
            alt_labels = {"User name": username} if language == "DE" else {"Benutzername": username}
            fill_result = await sap_fill_form_impl(alt_labels, strict=False)
            if fill_result.not_found:
                errors.append(f"Could not find username field: {fill_result.not_found}")

    # Set status checkboxes
    if statuses:
        checkbox_errors = await _set_status_checkboxes(page, statuses, language)
        errors.extend(checkbox_errors)

    # Fill date fields
    if from_date:
        sap_from = format_sap_date(from_date, language)
        from_labels = (
            {"Von Startdatum": sap_from} if language == "DE" else {"From start date": sap_from}
        )
        fill_result = await sap_fill_form_impl(from_labels, strict=False)
        if fill_result.not_found:
            alt_labels = (
                {"From start date": sap_from} if language == "DE" else {"Von Startdatum": sap_from}
            )
            fill_result = await sap_fill_form_impl(alt_labels, strict=False)

    if to_date:
        sap_to = format_sap_date(to_date, language)
        to_labels = (
            {"Bis Startdatum": sap_to} if language == "DE" else {"To start date": sap_to}
        )
        fill_result = await sap_fill_form_impl(to_labels, strict=False)
        if fill_result.not_found:
            alt_labels = (
                {"To start date": sap_to} if language == "DE" else {"Bis Startdatum": sap_to}
            )
            fill_result = await sap_fill_form_impl(alt_labels, strict=False)

    return errors


async def _fetch_job_log(page: Any, job: SM37Job, language: str) -> SM37JobLog | None:
    """
    Fetch the job log for a single job.

    The job must already be selected in the job list.

    Args:
        page: Playwright page object
        job: The job to fetch the log for
        language: SAP language ("DE" or "EN")

    Returns:
        SM37JobLog or None if log retrieval fails.
    """
    try:
        # Click the Jobprotokoll/Job log button
        log_button_text = "Jobprotokoll" if language == "DE" else "Job log"
        button = page.get_by_role("button", name=re.compile(re.escape(log_button_text), re.IGNORECASE))

        if await button.count() == 0:
            # Try menuitem as fallback (some SM37 versions use menu)
            button = page.get_by_role("menuitem", name=re.compile(re.escape(log_button_text), re.IGNORECASE))

        if await button.count() == 0:
            logger.warning("Job log button not found", extra={"label": log_button_text})
            return None

        await button.click()
        await page.wait_for_timeout(2000)
        await page.wait_for_load_state("networkidle")

        # Capture and parse the log screen
        snapshot = await page.locator("body").aria_snapshot()
        job_log = parse_sm37_job_log(snapshot, job.job_name, job.job_number)

        # Navigate back to job list (F3)
        await page.keyboard.press("F3")
        await page.wait_for_timeout(1000)

        return job_log

    except Exception as e:  # pylint: disable=broad-exception-caught
        logger.exception("Fetching job log", extra={"job": job.job_name})
        return None


async def _execute_sm37_lookup(
    job_name: str,
    username: str | None,
    statuses: list[str] | None,
    from_date: str | None,
    to_date: str | None,
    include_log: bool,
) -> SM37JobListResult:
    """
    Execute the SM37 lookup workflow.

    Args:
        job_name: Job name filter
        username: User filter
        statuses: Status filter list
        from_date: Start date (ISO)
        to_date: End date (ISO)
        include_log: Whether to fetch job log for single results

    Returns:
        SM37JobListResult with job data.
    """
    now = datetime.now(UTC)
    settings = get_settings()
    language = settings.sap_language

    # Navigate to SM37
    tx_result = await sap_transaction_impl("SM37")
    if not tx_result.success:
        return SM37JobListResult.failure(
            error=f"Failed to navigate to SM37: {tx_result.error}",
            jobs=[],
            job_count=0,
            filters_applied={},
            retrieved_at=now,
        )

    page = await (await get_browser_manager()).get_current_page()
    await page.wait_for_timeout(1000)

    # Fill selection screen
    fill_errors = await _fill_selection_screen(
        page, job_name, username, statuses, from_date, to_date, language
    )
    if fill_errors:
        logger.warning("Some selection fields had errors", extra={"errors": fill_errors})

    # Build filters summary
    filters_applied: dict[str, str] = {"job_name": job_name}
    if username:
        filters_applied["username"] = username
    if statuses:
        filters_applied["status"] = ",".join(statuses)
    if from_date:
        filters_applied["from_date"] = from_date
    if to_date:
        filters_applied["to_date"] = to_date

    # Execute (F8)
    await sap_press_key_impl("F8")
    await page.wait_for_timeout(3000)
    await page.wait_for_load_state("networkidle")

    # Get snapshot and check results
    snapshot = await page.locator("body").aria_snapshot()

    # Check for "no jobs found" message
    if is_no_jobs_found(snapshot):
        return SM37JobListResult(
            jobs=[],
            job_count=0,
            filters_applied=filters_applied,
            retrieved_at=datetime.now(UTC),
        )

    # Parse job list
    jobs = parse_sm37_job_list(snapshot)

    # Cap at MAX_JOBS
    if len(jobs) > MAX_JOBS:
        logger.warning("Truncating job list", extra={"total": len(jobs), "max": MAX_JOBS})
        jobs = jobs[:MAX_JOBS]

    # Fetch job log if requested and exactly one job
    job_log: SM37JobLog | None = None
    if include_log and len(jobs) == 1:
        # Select the first (only) row
        try:
            first_row = page.locator("[role='row']").nth(1)  # Skip header row
            if await first_row.count() > 0:
                await first_row.click()
                await page.wait_for_timeout(500)
                job_log = await _fetch_job_log(page, jobs[0], language)
        except Exception as e:  # pylint: disable=broad-exception-caught
            logger.warning("Could not select job row for log", extra={"error": str(e)})

    return SM37JobListResult(
        jobs=jobs,
        job_count=len(jobs),
        filters_applied=filters_applied,
        job_log=job_log,
        retrieved_at=datetime.now(UTC),
    )


# =============================================================================
# MCP Tool Registration
# =============================================================================


def register_sm37_tools(mcp: FastMCP) -> None:
    """Register SM37 tools with the MCP server."""

    @mcp.tool(
        annotations=ToolAnnotations(
            readOnlyHint=True,
            openWorldHint=False,
        ),
        description=(
            "List background jobs from SM37 (Job Overview). "
            "USE THIS instead of sap_transaction('SM37') - faster and returns structured data.\n\n"
            "Filters: job name (wildcards like *BILLING*), username, status "
            "(scheduled/released/active/finished/canceled), date range.\n\n"
            "When include_log=True and exactly one job matches, the job log is included.\n\n"
            "Returns job list with name, number, status, start/end times, duration, and user."
        ),
    )
    async def sap_sm37_lookup(
        job_name: str = "*",
        username: str | None = "*",
        status: list[Literal["scheduled", "released", "active", "finished", "canceled"]] | None = None,
        from_date: str | None = None,
        to_date: str | None = None,
        include_log: bool = False,
        output_file: str | None = None,
        session: str | None = None,
        agent_id: str | None = None,
    ) -> SM37JobListResult:
        """
        List background jobs from SM37.

        Args:
            job_name: Job name filter (supports * wildcard, default: *)
            username: User filter (default: * for all users, None to keep SAP default)
            status: Status filter list (default: None = all statuses).
                    Values: "scheduled", "released", "active", "finished", "canceled"
            from_date: Start date filter in ISO format (YYYY-MM-DD)
            to_date: End date filter in ISO format (YYYY-MM-DD)
            include_log: Fetch job log when exactly one job matches (default: False)
            output_file: If provided, write full results to this JSON file
            session: Session ID (e.g., "s1", "s2"). None uses primary session.
            agent_id: Agent identifier for binding check. Optional.

        Returns:
            SM37JobListResult with matching jobs and optional job log.
        """
        browser_manager = await get_browser_manager()

        try:
            browser_manager.get_session_page_checked(session, agent_id, "sap_sm37_lookup")
        except ValueError as e:
            return SM37JobListResult.failure(
                error=f"Session error: {e}",
                jobs=[],
                job_count=0,
                filters_applied={},
                retrieved_at=datetime.now(UTC),
            )

        result = await _execute_sm37_lookup(
            job_name=job_name,
            username=username,
            statuses=list(status) if status else None,
            from_date=from_date,
            to_date=to_date,
            include_log=include_log,
        )

        # Write to file if requested
        if output_file and result.success:
            output_path = Path(output_file)
            output_path.parent.mkdir(parents=True, exist_ok=True)

            with output_path.open("w", encoding="utf-8") as f:
                json.dump(result.model_dump(mode="json"), f, indent=2, ensure_ascii=False)

            logger.info("Wrote SM37 results to file", extra={"path": str(output_path)})

        return result
```

## Implementation Order

Execute tasks 1-10 in order. Tasks 1-4 have no SAP dependency and can be verified offline. Task 5 requires a live SAP system. Tasks 6-7 can be partially tested offline (date helper tests pass, parser tests skip until snapshots exist). Tasks 8-10 require SAP for full integration testing.

## Key Risks and Mitigations

1. **ARIA snapshot format unknown until exploration tests run.** The parser (Task 6) includes structural patterns but the exact row parsing regex will need refinement after Task 5 captures real snapshots. The parser is designed with this in mind -- column detection is dynamic, not hardcoded.

2. **Checkbox manipulation may fail in WebGUI.** SAP WebGUI checkboxes sometimes have non-standard ARIA roles. The tool uses `get_by_role("checkbox")` with fallback error reporting rather than hard-failing.

3. **Date field labels vary between SAP versions.** The tool tries both DE and EN labels as fallback, same pattern as SE16 tools.
