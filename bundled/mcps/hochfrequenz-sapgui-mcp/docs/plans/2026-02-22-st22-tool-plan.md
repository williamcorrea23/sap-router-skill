**Goal:** Add `sap_st22_lookup` read-only tool that lists short dumps by date and optionally retrieves dump detail text from ST22, returning strongly-typed Pydantic models.

**Architecture:** Single tool with two modes: (1) list dumps for a date (returns ALV grid rows as `ST22Dump` list), (2) select a specific dump by `dump_index` to retrieve detail text (`ST22DumpDetail` with best-effort section parsing + `raw_text` fallback). Parser module handles ALV list parsing and detail text extraction. Language constants in `lang.py` support DE/EN. Date format helper converts ISO dates to SAP locale format.

**Tech Stack:** Python 3.12, Playwright (browser automation), Pydantic v2 (models), FastMCP (tool registration), pytest + anyio (testing)

```python
"""Date format helpers for SAP locale conversion."""

from datetime import datetime

__all__ = ["format_sap_date"]


def format_sap_date(iso_date: str, language: str) -> str:
    """Convert ISO date (YYYY-MM-DD) to SAP locale format.

    Args:
        iso_date: Date in ISO format (YYYY-MM-DD)
        language: SAP language code ("DE" or "EN", case-insensitive)

    Returns:
        Formatted date string:
        - DE: DD.MM.YYYY
        - EN: MM/DD/YYYY

    Raises:
        ValueError: If iso_date is not valid YYYY-MM-DD format
    """
    dt = datetime.strptime(iso_date, "%Y-%m-%d")
    lang = language.upper()
    if lang == "DE":
        return dt.strftime("%d.%m.%Y")
    # EN and all other languages
    return dt.strftime("%m/%d/%Y")
```

```python
"""
Pydantic models for ST22 (Short Dump Analysis) lookup tool.

These models represent ABAP runtime error (short dump) data retrieved from ST22,
including dump list entries and detailed dump information.
"""

from pydantic import AwareDatetime, BaseModel, Field

from sapguimcp.models.base import ToolResult

__all__ = [
    "ST22Dump",
    "ST22DumpDetail",
    "ST22DumpListResult",
    "ST22DumpDetailResult",
]


class ST22Dump(BaseModel):
    """A short dump entry from the ST22 list view."""

    index: int = Field(description="Position in the list (0-based, for use with dump_index)")
    time: str = Field(description="Time of the dump (HH:MM:SS)")
    program: str = Field(description="ABAP program that caused the dump")
    include: str | None = Field(default=None, description="Include name (if applicable)")
    error_type: str = Field(description="Runtime error type (e.g., RABAX_STATE, MESSAGE_TYPE_X)")
    short_text: str = Field(description="Human-readable short description")
    user: str = Field(description="SAP user who triggered the dump")


class ST22DumpDetail(BaseModel):
    """Detailed information from a single short dump."""

    error_type: str = Field(description="Runtime error type")
    short_text: str = Field(description="Human-readable short description")
    what_happened: str = Field(default="", description="Extracted from 'What happened?' section")
    how_to_correct: str = Field(default="", description="Extracted from 'How to correct' section")
    program: str = Field(description="ABAP program name")
    include: str | None = Field(default=None, description="Include name")
    line: int | None = Field(default=None, description="Source line number where error occurred")
    call_stack: list[str] = Field(default_factory=list, description="Call stack entries")
    raw_text: str = Field(description="Full dump text (truncated to ~10KB) as fallback")


class ST22DumpListResult(ToolResult):
    """Result of ST22 dump list retrieval."""

    dumps: list[ST22Dump] = Field(default_factory=list, description="Short dump entries")
    dump_count: int = Field(default=0, description="Total number of dumps found")
    date_searched: str = Field(default="", description="Date that was searched (YYYY-MM-DD)")
    retrieved_at: AwareDatetime = Field(description="When the data was retrieved")


class ST22DumpDetailResult(ToolResult):
    """Result of ST22 dump detail retrieval."""

    detail: ST22DumpDetail | None = Field(default=None, description="Dump detail data")
    retrieved_at: AwareDatetime = Field(description="When the data was retrieved")
```

```python
"""
Unit tests for ST22 (Short Dump Analysis) parser.

Tests parsing of YAML accessibility snapshots from ST22 screens.
"""

from pathlib import Path

import pytest

from sapguimcp.models.st22_models import ST22Dump, ST22DumpDetail

SNAPSHOTS_DIR = Path(__file__).parent / "testdata" / "st22_exploration"


def _load_snapshot(name: str) -> str:
    """Load a YAML snapshot file."""
    filepath = SNAPSHOTS_DIR / f"{name}_de.yaml"
    if not filepath.exists():
        pytest.skip(f"Snapshot {filepath} not available - run exploration tests first")
    return filepath.read_text(encoding="utf-8")


class TestParseDumpList:
    """Tests for parsing ST22 dump list ALV grid."""

    def test_parse_dump_list_returns_dumps(self) -> None:
        """Dump list snapshot should parse into ST22Dump entries."""
        from sapguimcp.parsers.st22_parser import parse_st22_dump_list

        snapshot = _load_snapshot("st22_dump_list_grid")
        dumps = parse_st22_dump_list(snapshot)

        assert isinstance(dumps, list)
        # If the system had dumps, we should get at least one
        if len(dumps) > 0:
            dump = dumps[0]
            assert isinstance(dump, ST22Dump)
            assert dump.index == 0
            assert dump.time  # Non-empty
            assert dump.program  # Non-empty
            assert dump.error_type  # Non-empty
            assert dump.user  # Non-empty

    def test_dump_list_indexes_sequential(self) -> None:
        """Dump indexes should be sequential starting from 0."""
        from sapguimcp.parsers.st22_parser import parse_st22_dump_list

        snapshot = _load_snapshot("st22_dump_list_grid")
        dumps = parse_st22_dump_list(snapshot)

        for i, dump in enumerate(dumps):
            assert dump.index == i
```

```python
"""
Parser for ST22 (Short Dump Analysis) ARIA snapshots.

Extracts short dump list data from ALV grids and best-effort section
parsing from dump detail text pages. Falls back to raw_text for all
detail content.
"""

import logging
import re

from sapguimcp.lang import (
    ST22_COL_ERROR_DE,
    ST22_COL_ERROR_EN,
    ST22_COL_INCLUDE_DE,
    ST22_COL_INCLUDE_EN,
    ST22_COL_PROGRAM_DE,
    ST22_COL_PROGRAM_EN,
    ST22_COL_SHORT_TEXT_DE,
    ST22_COL_SHORT_TEXT_EN,
    ST22_COL_TIME_DE,
    ST22_COL_TIME_EN,
    ST22_COL_USER_DE,
    ST22_COL_USER_EN,
    ST22_HOW_TO_CORRECT_DE,
    ST22_HOW_TO_CORRECT_EN,
    ST22_WHAT_HAPPENED_DE,
    ST22_WHAT_HAPPENED_EN,
    bilingual_pattern,
)
from sapguimcp.models.st22_models import ST22Dump, ST22DumpDetail

logger = logging.getLogger(__name__)

__all__ = [
    "parse_st22_dump_list",
    "parse_st22_dump_detail",
]

# Max raw_text size (~10KB)
_MAX_RAW_TEXT_LENGTH = 10240

# =============================================================================
# ALV Grid Dump List Parsing
# =============================================================================

# Column header patterns for identifying columns by position
_COLUMNHEADER_PATTERN = re.compile(r'columnheader "(?P<name>[^"]+)"')

# Grid row pattern - matches data rows in ALV grid
_GRIDCELL_PATTERN = re.compile(r'gridcell "(?P<value>[^"]*)"')


def _identify_column_mapping(header_line: str) -> dict[int, str]:
    """Map column index to semantic field name from header row."""
    mapping: dict[int, str] = {}
    for i, match in enumerate(_COLUMNHEADER_PATTERN.finditer(header_line)):
        col_name = match.group("name").strip()
        col_lower = col_name.lower()

        if col_lower in (ST22_COL_TIME_DE.lower(), ST22_COL_TIME_EN.lower()):
            mapping[i] = "time"
        elif col_lower in (ST22_COL_PROGRAM_DE.lower(), ST22_COL_PROGRAM_EN.lower()):
            mapping[i] = "program"
        elif col_lower in (ST22_COL_INCLUDE_DE.lower(), ST22_COL_INCLUDE_EN.lower()):
            mapping[i] = "include"
        elif col_lower in (ST22_COL_ERROR_DE.lower(), ST22_COL_ERROR_EN.lower()):
            mapping[i] = "error_type"
        elif col_lower in (ST22_COL_SHORT_TEXT_DE.lower(), ST22_COL_SHORT_TEXT_EN.lower()):
            mapping[i] = "short_text"
        elif col_lower in (ST22_COL_USER_DE.lower(), ST22_COL_USER_EN.lower()):
            mapping[i] = "user"

    return mapping


def parse_st22_dump_list(snapshot: str) -> list[ST22Dump]:
    """
    Parse the ST22 dump list from an ALV grid ARIA snapshot.

    Args:
        snapshot: YAML accessibility snapshot from the ST22 list screen

    Returns:
        List of ST22Dump entries with sequential indexes
    """
    dumps: list[ST22Dump] = []
    lines = snapshot.split("\n")

    # Find header row and build column mapping
    column_mapping: dict[int, str] = {}
    header_found = False

    for line in lines:
        if "columnheader" in line and not header_found:
            # This line (or block) contains the header row
            column_mapping = _identify_column_mapping(line)
            if column_mapping:
                header_found = True
            continue

        # Parse data rows once we have column mapping
        if header_found and "gridcell" in line:
            cells = _GRIDCELL_PATTERN.findall(line)
            if not cells:
                continue

            row_data: dict[str, str] = {}
            for idx, value in enumerate(cells):
                field = column_mapping.get(idx)
                if field:
                    row_data[field] = value.strip()

            # Only create a dump if we have minimum required fields
            if "error_type" in row_data or "program" in row_data:
                dumps.append(
                    ST22Dump(
                        index=len(dumps),
                        time=row_data.get("time", ""),
                        program=row_data.get("program", ""),
                        include=row_data.get("include") or None,
                        error_type=row_data.get("error_type", ""),
                        short_text=row_data.get("short_text", ""),
                        user=row_data.get("user", ""),
                    )
                )

    return dumps


# =============================================================================
# Dump Detail Text Parsing
# =============================================================================

def _extract_text_content(snapshot: str) -> str:
    """Extract readable text content from ARIA snapshot, stripping ARIA markup."""
    # Extract all text content from the snapshot
    text_parts: list[str] = []
    # Match text content in quotes (from various ARIA elements)
    text_pattern = re.compile(r'"([^"]{2,})"')
    for match in text_pattern.finditer(snapshot):
        text_parts.append(match.group(1))
    return "\n".join(text_parts)


def _extract_section(text: str, start_markers: list[str], end_markers: list[str]) -> str:
    """Extract text between section markers (best-effort)."""
    text_lower = text.lower()
    start_pos = -1
    for marker in start_markers:
        pos = text_lower.find(marker.lower())
        if pos >= 0:
            start_pos = pos + len(marker)
            break

    if start_pos < 0:
        return ""

    # Find the next section header
    end_pos = len(text)
    for marker in end_markers:
        pos = text_lower.find(marker.lower(), start_pos)
        if 0 < pos < end_pos:
            end_pos = pos

    return text[start_pos:end_pos].strip()


def _extract_error_type(text: str) -> str:
    """Extract error type from dump detail text."""
    # Look for pattern like "Runtime Error: RABAX_STATE" or "Laufzeitfehler: RABAX_STATE"
    pattern = re.compile(r"(?:Runtime Error|Laufzeitfehler)[:\s]+([A-Z_0-9]+)", re.IGNORECASE)
    match = pattern.search(text)
    return match.group(1) if match else ""


def _extract_program(text: str) -> str:
    """Extract program name from dump detail text."""
    pattern = re.compile(r"(?:Program|Programm)[:\s]+([A-Z0-9_/]+)", re.IGNORECASE)
    match = pattern.search(text)
    return match.group(1) if match else ""


def _extract_line_number(text: str) -> int | None:
    """Extract source line number from dump detail text."""
    pattern = re.compile(r"(?:Source Code Line|Zeile im Quelltext)[:\s]+(\d+)", re.IGNORECASE)
    match = pattern.search(text)
    return int(match.group(1)) if match else None


def _extract_call_stack(text: str) -> list[str]:
    """Extract call stack entries from dump detail text (best-effort)."""
    # Look for call stack section
    stack_markers = ["call stack", "aufrufhierarchie", "active calls"]
    text_lower = text.lower()
    start_pos = -1
    for marker in stack_markers:
        pos = text_lower.find(marker)
        if pos >= 0:
            start_pos = pos
            break

    if start_pos < 0:
        return []

    # Extract lines that look like stack entries (contain line numbers or program names)
    lines = text[start_pos:start_pos + 2000].split("\n")
    stack: list[str] = []
    for line in lines[1:]:  # Skip the header line
        stripped = line.strip()
        if not stripped:
            continue
        # Stack entries typically contain program names and line numbers
        if re.search(r"[A-Z_/]+.*\d+", stripped):
            stack.append(stripped)
        elif len(stack) > 0 and not stripped:
            break  # Empty line after stack entries = end of stack

    return stack[:50]  # Cap at 50 entries


def parse_st22_dump_detail(snapshot: str) -> ST22DumpDetail:
    """
    Parse ST22 dump detail from an ARIA snapshot.

    Best-effort parsing: extracts known sections when possible,
    always provides raw_text as fallback.

    Args:
        snapshot: YAML accessibility snapshot from the ST22 dump detail screen

    Returns:
        ST22DumpDetail with extracted data and raw_text fallback
    """
    # Extract readable text
    raw_text = _extract_text_content(snapshot)
    if len(raw_text) > _MAX_RAW_TEXT_LENGTH:
        raw_text = raw_text[:_MAX_RAW_TEXT_LENGTH] + "\n... [truncated]"

    # Best-effort section extraction
    what_happened = _extract_section(
        raw_text,
        [ST22_WHAT_HAPPENED_DE, ST22_WHAT_HAPPENED_EN],
        [ST22_HOW_TO_CORRECT_DE, ST22_HOW_TO_CORRECT_EN, "Error Analysis", "Fehleranalyse"],
    )

    how_to_correct = _extract_section(
        raw_text,
        [ST22_HOW_TO_CORRECT_DE, ST22_HOW_TO_CORRECT_EN],
        ["Error Analysis", "Fehleranalyse", "Source Code", "Quelltext"],
    )

    return ST22DumpDetail(
        error_type=_extract_error_type(raw_text),
        short_text="",  # Will be populated from list data or extracted
        what_happened=what_happened,
        how_to_correct=how_to_correct,
        program=_extract_program(raw_text),
        include=None,  # Hard to extract reliably
        line=_extract_line_number(raw_text),
        call_stack=_extract_call_stack(raw_text),
        raw_text=raw_text,
    )
```

```python
"""
ST22 (Short Dump Analysis) lookup tool.

This module provides a tool to list and inspect ABAP runtime errors (short dumps)
from ST22, returning strongly-typed Pydantic models with dump details.

import json

async def _navigate_to_st22(page: Any) -> str | None:
    """Navigate to ST22. Returns error message or None on success."""
    tx_result = await sap_transaction_impl("ST22")
    if not tx_result.success:
        return f"Failed to navigate to ST22: {tx_result.error}"

    await page.wait_for_timeout(500)
    await page.wait_for_load_state("networkidle")
    await page.wait_for_timeout(500)
    return None


async def _select_date(page: Any, target_date: str | None) -> str | None:
    """
    Select the date for dump listing.

    Uses Today/Yesterday toolbar buttons when applicable, otherwise
    enters date into the date field.

    Args:
        page: Playwright page
        target_date: ISO date string (YYYY-MM-DD) or None for today

    Returns:
        Error message or None on success
    """
    settings = get_settings()
    language = settings.sap_language

    today = date.today()

    if target_date is None or target_date == today.isoformat():
        # Click "Today" button
        today_labels = ["Heute", "Today"]
        for label in today_labels:
            btn = page.get_by_role("button", name=label)
            if await btn.count() > 0:
                await btn.click()
                await page.wait_for_timeout(1000)
                await page.wait_for_load_state("networkidle")
                return None
        # If no Today button found, we may already be on today's view
        return None

    yesterday = (today - __import__("datetime").timedelta(days=1))
    if target_date == yesterday.isoformat():
        # Click "Yesterday" button
        yesterday_labels = ["Gestern", "Yesterday"]
        for label in yesterday_labels:
            btn = page.get_by_role("button", name=label)
            if await btn.count() > 0:
                await btn.click()
                await page.wait_for_timeout(1000)
                await page.wait_for_load_state("networkidle")
                return None
        # Fall through to date input if button not found

    # For other dates, use the date input field
    # Import the date helper
    from sapguimcp.utils.date_helpers import format_sap_date

    try:
        formatted_date = format_sap_date(target_date, language)
    except ValueError:
        return f"Invalid date format: {target_date} (expected YYYY-MM-DD)"

    # Find and fill the date field
    # NOTE: Exact field selector must be determined from exploration snapshots
    date_field = page.locator("input[type='text']").first
    if await date_field.count() > 0:
        await date_field.click(click_count=3)
        await page.wait_for_timeout(100)
        await page.keyboard.press("Delete")
        await page.keyboard.type(formatted_date)
        await page.keyboard.press("Enter")
        await page.wait_for_timeout(1000)
        await page.wait_for_load_state("networkidle")
        return None

    return f"Could not find date field to enter date {target_date}"


async def _select_dump_by_index(page: Any, dump_index: int, dump_count: int) -> str | None:
    """
    Select a dump from the list by double-clicking the row at dump_index.

    Args:
        page: Playwright page
        dump_index: 0-based index of the dump to select
        dump_count: Total dumps in the list (for bounds checking)

    Returns:
        Error message or None on success
    """
    if dump_index < 0 or dump_index >= dump_count:
        return f"dump_index {dump_index} out of range (0..{dump_count - 1})"

    # Select the row in the ALV grid
    # NOTE: Exact row selection method must be determined from exploration snapshots
    # Common approach: find all data rows and click the Nth one
    rows = page.locator("[role='row']")
    row_count = await rows.count()

    # Skip header row (index 0)
    target_row_idx = dump_index + 1  # +1 for header row
    if target_row_idx >= row_count:
        return f"Row index {target_row_idx} exceeds visible rows ({row_count})"

    row = rows.nth(target_row_idx)
    await row.dblclick()
    await page.wait_for_timeout(2000)
    await page.wait_for_load_state("networkidle")
    await page.wait_for_timeout(500)

    return None


async def _capture_full_detail(page: Any) -> str:
    """
    Capture the full dump detail by scrolling and collecting snapshots.

    ST22 detail is a long scrollable text. Scroll down and concatenate
    snapshots to get as much content as possible.
    """
    snapshots: list[str] = []

    # Capture initial view
    snapshot = await page.locator("body").aria_snapshot()
    snapshots.append(snapshot)

    # Scroll down to capture more content (up to 10 pages)
    for _ in range(10):
        await page.keyboard.press("PageDown")
        await page.wait_for_timeout(300)
        new_snapshot = await page.locator("body").aria_snapshot()

        # Stop if we get the same content (reached bottom)
        if new_snapshot == snapshots[-1]:
            break
        snapshots.append(new_snapshot)

    return "\n".join(snapshots)

async def _st22_lookup(
    page: Any,
    target_date: str | None,
    username: str | None,
    program: str | None,
    dump_index: int | None,
) -> ST22DumpListResult | ST22DumpDetailResult:
    """Core ST22 lookup logic."""
    now = datetime.now(UTC)
    date_str = target_date or date.today().isoformat()

    # Navigate to ST22
    error = await _navigate_to_st22(page)
    if error:
        return ST22DumpListResult.failure(
            error=error,
            dumps=[],
            dump_count=0,
            date_searched=date_str,
            retrieved_at=now,
        )

    # Select the target date
    error = await _select_date(page, target_date)
    if error:
        return ST22DumpListResult.failure(
            error=error,
            dumps=[],
            dump_count=0,
            date_searched=date_str,
            retrieved_at=now,
        )

    # Wait for dump list to load
    await page.wait_for_timeout(500)

    # Capture dump list snapshot
    list_snapshot = await page.locator("body").aria_snapshot()
    logger.debug("ST22 list snapshot captured", extra={"length": len(list_snapshot)})

    # Parse the dump list
    dumps = parse_st22_dump_list(list_snapshot)

    # Apply client-side filters (username, program)
    if username:
        dumps = [d for d in dumps if d.user.upper() == username.upper()]
    if program:
        program_upper = program.upper()
        dumps = [d for d in dumps if program_upper in d.program.upper()]

    # Re-index after filtering
    for i, dump in enumerate(dumps):
        dump.index = i

    # If no dump_index requested, return the list
    if dump_index is None:
        return ST22DumpListResult(
            dumps=dumps,
            dump_count=len(dumps),
            date_searched=date_str,
            retrieved_at=now,
        )

    # Select specific dump for detail view
    if not dumps:
        return ST22DumpDetailResult.failure(
            error="No dumps found to select from",
            detail=None,
            retrieved_at=now,
        )

    # Bounds check
    if dump_index < 0 or dump_index >= len(dumps):
        return ST22DumpDetailResult.failure(
            error=f"dump_index {dump_index} out of range (0..{len(dumps) - 1})",
            detail=None,
            retrieved_at=now,
        )

    # Need to map filtered index back to original row position
    # Re-parse unfiltered list to get actual grid positions
    all_dumps = parse_st22_dump_list(list_snapshot)
    target_dump = dumps[dump_index]

    # Find matching dump in unfiltered list by time+program+error_type
    original_index = None
    for d in all_dumps:
        if d.time == target_dump.time and d.program == target_dump.program and d.error_type == target_dump.error_type:
            original_index = d.index
            break

    if original_index is None:
        original_index = dump_index  # Fallback

    error = await _select_dump_by_index(page, original_index, len(all_dumps))
    if error:
        return ST22DumpDetailResult.failure(
            error=error,
            detail=None,
            retrieved_at=now,
        )

    # Capture detail by scrolling through pages
    detail_snapshot = await _capture_full_detail(page)

    # Parse the detail
    detail = parse_st22_dump_detail(detail_snapshot)

    # Enrich detail with data from the list entry
    detail.short_text = target_dump.short_text
    if not detail.program:
        detail.program = target_dump.program
    if not detail.error_type:
        detail.error_type = target_dump.error_type

    return ST22DumpDetailResult(
        detail=detail,
        retrieved_at=now,
    )

def register_st22_tools(mcp: FastMCP) -> None:
    """Register ST22 tools with the MCP server."""

    @mcp.tool(
        annotations=ToolAnnotations(
            readOnlyHint=True,
            openWorldHint=False,
        ),
        description=(
            "Look up ABAP short dumps (runtime errors) from ST22. "
            "USE THIS instead of sap_transaction('ST22') - faster and returns structured data. "
            "By default lists today's dumps. Use date parameter for other dates. "
            "Set dump_index to retrieve full detail text for a specific dump. "
            "Workflow: call once without dump_index to get the list, then call again "
            "with dump_index to read a specific dump's detail."
        ),
    )
    async def sap_st22_lookup(
        date: str | None = None,
        username: str | None = None,
        program: str | None = None,
        dump_index: int | None = None,
        output_file: str | None = None,
        session: str | None = None,
        agent_id: str | None = None,
    ) -> ST22DumpListResult | ST22DumpDetailResult:
        """
        Look up short dumps from ST22.

        Args:
            date: Date to search (YYYY-MM-DD). Default: today.
                  Uses Today/Yesterday toolbar buttons when applicable.
            username: Filter dumps by SAP user (case-insensitive).
            program: Filter dumps by ABAP program name (substring match, case-insensitive).
            dump_index: Select the Nth dump from the list (0-based) to read detail.
                        If None, returns the dump list only.
            output_file: If provided, write results to this JSON file and return summary.
            session: Session ID (e.g., "s1", "s2"). None uses primary session.
            agent_id: Agent identifier for binding check. Optional.

        Returns:
            ST22DumpListResult (when dump_index is None) with list of dumps, or
            ST22DumpDetailResult (when dump_index is set) with full dump details
        """
        browser_manager = await get_browser_manager()

        try:
            page = browser_manager.get_session_page_checked(session, agent_id, "sap_st22_lookup")
        except ValueError as e:
            now = datetime.now(UTC)
            return ST22DumpListResult.failure(
                error=f"Session error: {e}",
                dumps=[],
                dump_count=0,
                date_searched=date or "",
                retrieved_at=now,
            )

        try:
            result = await _st22_lookup(page, date, username, program, dump_index)
        except Exception as e:
            logger.exception("ST22 lookup failed")
            now = datetime.now(UTC)
            return ST22DumpListResult.failure(
                error=f"ST22 lookup failed: {e}",
                dumps=[],
                dump_count=0,
                date_searched=date or "",
                retrieved_at=now,
            )

        # Write to file if requested
        if output_file:
            output_path = Path(output_file)
            output_path.parent.mkdir(parents=True, exist_ok=True)

            with output_path.open("w", encoding="utf-8") as f:
                json.dump(result.model_dump(mode="json"), f, indent=2, ensure_ascii=False)

            # For file output, return the result as-is (it already has success/error)
            # The caller can read the file for details

        return result
```

```python
    """Get detail of the first dump (if any exist)."""
    login = await call_tool_typed(sap_mcp_client, "sap_login", {}, LoginResult)
```

## Key Design Decisions

1. **`dump_index` not `dump_id`**: ST22 dumps lack a single unique ID. The tool lists dumps, the caller selects by 0-based index.
2. **`raw_text` fallback**: ST22 detail pages vary widely. Best-effort section parsing plus full raw text ensures the LLM always gets useful data.
3. **No `variables` in v1**: Variable values in ST22 are deeply nested across multiple pages. The `raw_text` field provides access to all content.
4. **Client-side filtering**: Username and program filters are applied after parsing the full list, not via SAP selection screen fields (ST22 has limited selection screen).
5. **Date navigation**: Today/Yesterday buttons preferred over date field input for reliability.
6. **Exploration tests first**: Parser patterns depend on real YAML snapshots. Always run exploration tests before finalizing parser regex.
