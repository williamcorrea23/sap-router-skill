**Goal:** Implement a read-only SM30 Table Maintenance View lookup tool that dynamically parses flat table views from ARIA snapshots and returns structured row data.
**Architecture:** SM30 follows the established SE37/SE93 lookup pattern: Pydantic models (`sm30_models.py`), a dedicated parser (`sm30_parser.py`) for extracting dynamic column headers and row values from ARIA snapshots, and a tool module (`sm30_tools.py`) registered with FastMCP.
```python
"""
Pydantic models for SM30 (Table Maintenance View) lookup tool.

These models represent view data retrieved from SM30 in display mode,
including dynamically-parsed column headers and row values.
"""

from typing import Any, Literal

from pydantic import AwareDatetime, BaseModel, Field

from sapguimcp.models.base import ToolResult

__all__ = [
    "SM30Row",
    "SM30ViewResult",
    "SM30FileSummary",
]

# View type detection
SM30ViewType = Literal["flat", "unsupported"]


class SM30Row(BaseModel):
    """A single row from an SM30 table maintenance view.

    Uses dict[str, str] since SM30 views contain configuration data
    that is best represented as strings (unlike SE16 which benefits
    from type coercion for numeric queries).
    """

    values: dict[str, str] = Field(description="Column name -> cell value")


class SM30ViewResult(ToolResult):
    """Result of SM30 table maintenance view lookup."""

    view_name: str = Field(description="View or table name that was looked up")
    description: str = Field(default="", description="View description from screen title")
    view_type: SM30ViewType = Field(description="Detected view type: 'flat' or 'unsupported'")
    columns: list[str] = Field(default_factory=list, description="Column names in display order")
    rows: list[SM30Row] = Field(default_factory=list, description="All rows from the view")
    row_count: int = Field(default=0, description="Number of rows returned")
    retrieved_at: AwareDatetime = Field(description="UTC timestamp when data was retrieved")


class SM30FileSummary(ToolResult):
    """Summary result when SM30 output is written to file.

    Returned instead of SM30ViewResult when output_file parameter is provided.
    Contains metadata and a preview of the first few rows.
    """

    output_file: str = Field(description="Path to JSON file containing full SM30ViewResult")
    view_name: str = Field(description="View or table name that was looked up")
    description: str = Field(default="", description="View description")
    view_type: SM30ViewType = Field(description="Detected view type")
    columns: list[str] = Field(default_factory=list, description="Column names in order")
    row_count: int = Field(default=0, description="Total rows written to file")
    sample_rows: list[SM30Row] = Field(default_factory=list, description="Preview of first 5 rows")
```

### Task 5: Write SM30 Parser (`sm30_parser.py`)

**Files:** Create `src/sapguimcp/parsers/sm30_parser.py`

**Step 1:** Create the parser. SM30 display mode renders a standard ALV-like grid. The parser must:

1. Detect if the view loaded successfully (vs. not found / SM34 redirect)
2. Extract column headers dynamically from `columnheader` elements
3. Extract rows from `gridcell` elements
4. Handle variable column counts (3 to 20+)

```python
"""
Parser for SM30 (Table Maintenance View) ARIA snapshots.

Extracts structured row data from SM30 display mode grids, handling:
- Dynamic column headers (views have 3 to 20+ columns)
- Data rows with gridcell values
- View type detection (flat vs unsupported/SM34)
- German and English label support
- View not found detection
"""

import logging
import re
from datetime import UTC, datetime

from sapguimcp.lang import (
    SM30_NOT_FOUND_DE,
    SM30_NOT_FOUND_EN,
    SM30_ROW_SELECT_HINT_DE,
    SM30_ROW_SELECT_HINT_EN,
    SM30_USE_SM34_DE,
    SM30_USE_SM34_EN,
)
from sapguimcp.models.sm30_models import SM30Row, SM30ViewResult, SM30ViewType

logger = logging.getLogger(__name__)

__all__ = [
    "parse_sm30_snapshot",
    "parse_sm30_columns",
    "parse_sm30_rows",
    "detect_sm30_view_type",
]

# =============================================================================
# Regex Patterns
# =============================================================================

# Column header extraction from grid
_COLUMNHEADER_PATTERN = re.compile(r'columnheader "(?P<name>[^"]+)"')

# Header row pattern (row containing columnheader elements)
_HEADER_ROW_PATTERN = re.compile(
    r'- row "[^"]*":\s*\n(?:\s*- columnheader "[^"]+"[^\n]*\n?)+',
    re.MULTILINE,
)

# Data row pattern - SM30 display mode rows
# SM30 data rows start with "To select..." or "Zum Auswählen..." hint text
_ROW_START_PATTERN = re.compile(
    rf'- \'?row "(?:{re.escape(SM30_ROW_SELECT_HINT_DE)}|{re.escape(SM30_ROW_SELECT_HINT_EN)})'
)

# Gridcell value extraction
_GRIDCELL_WITH_VALUE_PATTERN = re.compile(r'gridcell "(?P<value>[^"]*)"')
_GRIDCELL_EMPTY_PATTERN = re.compile(r"^\s*- gridcell:\s*$")

# Row selection column names to skip
_SELECTION_COLUMN_NAMES = {
    "Spalte für Zeilenauswahl",
    "Column for row selection",
}

# SM34 detection patterns
_SM34_PATTERN = re.compile(
    rf"(?:{re.escape(SM30_USE_SM34_DE)}|{re.escape(SM30_USE_SM34_EN)})",
    re.IGNORECASE,
)

# View not found patterns
_NOT_FOUND_PATTERNS = [
    SM30_NOT_FOUND_DE,
    SM30_NOT_FOUND_EN,
    "does not exist",
    "existiert nicht",
    "nicht vorhanden",
    "not found",
]

# View description from page title
# DE: 'heading "Sicht "Länder" anzeigen: Übersicht"'
# EN: 'heading "Display View "Countries": Overview"'
_VIEW_TITLE_PATTERN = re.compile(
    r'heading "(?:Sicht|Display View)\s+"(?P<desc>[^"]+)"\s+(?:anzeigen|display)',
    re.IGNORECASE,
)

# Alternative: extract from generic heading
_HEADING_PATTERN = re.compile(r'heading "(?P<title>[^"]+)"')


# =============================================================================
# Detection Functions
# =============================================================================


def detect_sm30_view_type(snapshot: str) -> SM30ViewType:
    """
    Detect whether the SM30 snapshot shows a flat table or an unsupported view type.

    Returns "flat" if a grid with columnheaders is found.
    Returns "unsupported" if SM34 redirect is detected or no grid found.
    """
    # Check for SM34 redirect message
    if _SM34_PATTERN.search(snapshot):
        return "unsupported"

    # Check for grid with column headers (indicates flat table)
    if _COLUMNHEADER_PATTERN.search(snapshot):
        return "flat"

    return "unsupported"


def _is_view_not_found(snapshot: str) -> bool:
    """Check if the snapshot indicates the view was not found."""
    snapshot_lower = snapshot.lower()
    return any(pattern.lower() in snapshot_lower for pattern in _NOT_FOUND_PATTERNS)


def _extract_view_description(snapshot: str) -> str:
    """Extract the view description from the page heading."""
    match = _VIEW_TITLE_PATTERN.search(snapshot)
    if match:
        return match.group("desc")

    # Fallback: use first heading
    match = _HEADING_PATTERN.search(snapshot)
    if match:
        return match.group("title")

    return ""


# =============================================================================
# Column Parsing
# =============================================================================


def parse_sm30_columns(snapshot: str) -> list[str]:
    """
    Extract column names from the SM30 grid header row.

    Skips the row selection column (present in display mode).
    Handles variable column counts dynamically.

    Args:
        snapshot: ARIA snapshot YAML content

    Returns:
        List of column names in order
    """
    columns: list[str] = []

    # Find the header row section
    header_match = _HEADER_ROW_PATTERN.search(snapshot)
    if header_match:
        header_section = header_match.group(0)
    else:
        # Fallback: search from grid start to first data row
        grid_start = snapshot.find("- grid:")
        if grid_start == -1:
            grid_start = snapshot.find("grid:")
        if grid_start == -1:
            logger.warning("No grid found in SM30 snapshot")
            return columns

        # Find first data row
        first_row_de = snapshot.find(f'row "{SM30_ROW_SELECT_HINT_DE}', grid_start)
        first_row_en = snapshot.find(f'row "{SM30_ROW_SELECT_HINT_EN}', grid_start)

        candidates = [pos for pos in [first_row_de, first_row_en] if pos != -1]
        first_row = min(candidates) if candidates else len(snapshot)

        header_section = snapshot[grid_start:first_row]

    # Extract all columnheader values
    for match in _COLUMNHEADER_PATTERN.finditer(header_section):
        col_name = match.group("name")
        # Skip the row selection column
        if col_name not in _SELECTION_COLUMN_NAMES:
            columns.append(col_name)

    return columns


# =============================================================================
# Row Parsing
# =============================================================================


def _extract_gridcell_value(line: str) -> str | None:
    """Extract value from a gridcell line, returning None if not a gridcell."""
    match = _GRIDCELL_WITH_VALUE_PATTERN.search(line)
    if match:
        return match.group("value")
    if _GRIDCELL_EMPTY_PATTERN.match(line):
        return ""
    return None


def parse_sm30_rows(snapshot: str, columns: list[str] | None = None) -> list[SM30Row]:
    """
    Extract data rows from SM30 display grid snapshot.

    Each row is returned as an SM30Row with values dict mapping column name to value.

    Args:
        snapshot: ARIA snapshot YAML content
        columns: Column names (if None, will be extracted from snapshot)

    Returns:
        List of SM30Row instances
    """
    if columns is None:
        columns = parse_sm30_columns(snapshot)

    if not columns:
        logger.warning("No columns found, cannot parse SM30 rows")
        return []

    rows: list[SM30Row] = []
    lines = snapshot.split("\n")
    in_data_row = False
    current_cells: list[str] = []

    for line in lines:
        # Check for start of a data row
        if _ROW_START_PATTERN.search(line):
            # Save previous row if valid
            _save_row_if_valid(current_cells, columns, rows)
            current_cells = []
            in_data_row = True
            continue

        if in_data_row:
            # Check for end of data section
            if "- rowgroup" in line or "- menu:" in line:
                _save_row_if_valid(current_cells, columns, rows)
                in_data_row = False
                current_cells = []
                continue

            # Extract gridcell value
            cell_value = _extract_gridcell_value(line)
            if cell_value is not None:
                current_cells.append(cell_value)

    # Don't forget the last row
    if in_data_row:
        _save_row_if_valid(current_cells, columns, rows)

    return rows


def _save_row_if_valid(
    current_cells: list[str],
    columns: list[str],
    rows: list[SM30Row],
) -> None:
    """Save current cells as an SM30Row if valid (has data beyond selection cell)."""
    if current_cells and len(current_cells) > 1:
        # Skip first cell (row selection column)
        data_cells = current_cells[1:]
        values: dict[str, str] = {}
        for i, col_name in enumerate(columns):
            if i < len(data_cells):
                values[col_name] = data_cells[i]
            else:
                values[col_name] = ""
        if values:
            rows.append(SM30Row(values=values))


# =============================================================================
# Main Parser
# =============================================================================


def parse_sm30_snapshot(
    snapshot: str,
    view_name: str,
) -> SM30ViewResult:
    """
    Parse SM30 display snapshot into structured data.

    This is the main entry point. Handles:
    - View not found
    - SM34 redirect detection
    - Flat table parsing (dynamic columns + rows)

    Args:
        snapshot: The YAML accessibility snapshot from browser_snapshot
        view_name: The view/table name being looked up

    Returns:
        SM30ViewResult with parsed data or error
    """
    now = datetime.now(UTC)

    # Check for view not found
    if _is_view_not_found(snapshot):
        return SM30ViewResult.failure(
            error=f"View '{view_name}' not found",
            view_name=view_name,
            description="",
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Detect view type
    view_type = detect_sm30_view_type(snapshot)

    if view_type == "unsupported":
        return SM30ViewResult.failure(
            error=(
                f"View '{view_name}' requires SM34 (Extended Table Maintenance) or has "
                "an unsupported layout. Only flat table views are supported."
            ),
            view_name=view_name,
            description=_extract_view_description(snapshot),
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Parse flat table
    description = _extract_view_description(snapshot)
    columns = parse_sm30_columns(snapshot)
    rows = parse_sm30_rows(snapshot, columns)

    return SM30ViewResult(
        view_name=view_name,
        description=description,
        view_type="flat",
        columns=columns,
        rows=rows,
        row_count=len(rows),
        retrieved_at=now,
    )
```

```python
SM30 (Table Maintenance View) read-only lookup tool.

This module provides a tool to display SM30 table maintenance views,
returning structured data with dynamically-parsed columns and rows.

__all__ = ["register_sm30_tools"]

async def _find_view_field(page: Any) -> Any:
    """Find the Table/View input field in SM30 using multiple strategies."""
    strategies = [
        # Strategy 1: Try by role with exact name matches (DE/EN)
        page.get_by_role("textbox", name="Tabelle/Sicht"),
        page.get_by_role("textbox", name="Table/View"),
        # Strategy 2: Try regex pattern
        page.get_by_role(
            "textbox",
            name=re.compile(r"Tabelle/Sicht|Table/View", re.I),
        ),
        # Strategy 3: Try by title attribute
        page.locator(
            "input[title*='Tabelle'], input[title*='Table/View']"
        ).first,
        # Strategy 4: First visible input field (last resort)
        page.locator("input:visible").first,
    ]

    for field in strategies:
        if await field.count() > 0:
            return field

    return None


async def _fill_view_field(page: Any, view_name: str) -> str | None:
    """Fill the view name field in SM30. Returns error string or None."""
    view_field = await _find_view_field(page)

    if view_field is None or await view_field.count() == 0:
        return "Could not find Table/View field in SM30"

    # Clear the field
    await view_field.click(click_count=3)
    await page.wait_for_timeout(100)
    await page.keyboard.press("Delete")
    await page.wait_for_timeout(50)

    # Type the view name
    await page.keyboard.type(view_name.upper())
    await page.wait_for_timeout(100)
    return None


async def _ensure_display_mode(page: Any) -> None:
    """
    Ensure the Display radio button is selected (not Maintain).

    SM30 initial screen has two radio buttons:
    - DE: "Anzeigen" / EN: "Display"
    - DE: "Pflegen" / EN: "Maintain"

    We must select Display for read-only access.
    """
    # Try to find and click the Display radio button
    for label in ["Anzeigen", "Display"]:
        radio = page.get_by_role("radio", name=label)
        if await radio.count() > 0:
            await radio.click()
            await page.wait_for_timeout(200)
            return

    # If radio buttons not found, just proceed (F7 forces display mode anyway)
    logger.debug("Display radio button not found, proceeding with default")


async def _check_view_not_found(page: Any, view_name: str) -> str | None:
    """
    Check if the view was not found after pressing Enter/F7.

    Returns error message if not found, None if view loaded successfully.
    """
    # Check status bar for error messages
    status_bar = page.locator("#sapStatusBarAll, [id*='STATUSBAR']").first
    status_text = await status_bar.text_content() if await status_bar.count() > 0 else ""

    not_found_msgs = {
        "existiert nicht",
        "does not exist",
        "nicht vorhanden",
        "not found",
        "ist nicht vorhanden",
        "kein Pflegedialog",
        "no maintenance dialog",
    }

    if status_text and any(msg in status_text.lower() for msg in not_found_msgs):
        return f"View '{view_name}' not found: {status_text.strip()}"

    # Check page title - if still on initial screen, view wasn't found
    page_title = await page.title()
    is_initial = (
        "Tabellensicht-Pflege" in page_title
        or "Table View Maintenance" in page_title
        or "Einstieg" in page_title
        or "Initial" in page_title
    )

    if is_initial:
        # Still on initial screen
        if status_text:
            return f"View '{view_name}' not found: {status_text.strip()}"
        return f"View '{view_name}' not found (still on initial screen)"

    return None


async def _check_sm34_redirect(page: Any) -> str | None:
    """
    Check if SM30 redirected to SM34 (extended table maintenance).

    Returns error message if SM34 redirect detected, None otherwise.
    """
    # Check status bar for SM34 mention
    status_bar = page.locator("#sapStatusBarAll, [id*='STATUSBAR']").first
    status_text = await status_bar.text_content() if await status_bar.count() > 0 else ""

    if status_text and "sm34" in status_text.lower():
        return f"This view requires SM34 (Extended Table Maintenance): {status_text.strip()}"

    # Check snapshot for SM34 indicators
    snapshot = await page.locator("body").aria_snapshot()
    if "SM34" in snapshot:
        return "This view requires SM34 (Extended Table Maintenance)"

    return None

async def _lookup_view(page: Any, view_name: str) -> SM30ViewResult:
    """Look up a single SM30 view."""
    now = datetime.now(UTC)

    # Navigate to SM30
    tx_result = await sap_transaction_impl("SM30")
    if not tx_result.success:
        return SM30ViewResult.failure(
            error=f"Failed to navigate to SM30: {tx_result.error}",
            view_name=view_name,
            description="",
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Wait for SM30 screen to be ready
    await page.wait_for_timeout(500)
    await page.wait_for_load_state("networkidle")

    # Ensure display mode is selected
    await _ensure_display_mode(page)

    # Fill view name
    fill_error = await _fill_view_field(page, view_name)
    if fill_error:
        return SM30ViewResult.failure(
            error=fill_error,
            view_name=view_name,
            description="",
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Press Enter to display the view (Display button shortcut)
    await page.wait_for_timeout(300)
    await page.keyboard.press("Enter")
    await page.wait_for_timeout(1000)
    await page.wait_for_load_state("networkidle")
    await page.wait_for_timeout(500)

    # Check for view not found
    not_found_error = await _check_view_not_found(page, view_name)
    if not_found_error:
        return SM30ViewResult.failure(
            error=not_found_error,
            view_name=view_name,
            description="",
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Check for SM34 redirect
    sm34_error = await _check_sm34_redirect(page)
    if sm34_error:
        return SM30ViewResult.failure(
            error=sm34_error,
            view_name=view_name,
            description="",
            view_type="unsupported",
            columns=[],
            rows=[],
            row_count=0,
            retrieved_at=now,
        )

    # Get snapshot and parse
    snapshot = await page.locator("body").aria_snapshot()
    logger.debug(
        "Got SM30 snapshot",
        extra={"view": view_name, "length": len(snapshot)},
    )

    return parse_sm30_snapshot(snapshot, view_name)

def register_sm30_tools(mcp: FastMCP) -> None:
    """Register SM30 tools with the MCP server."""

    @mcp.tool(
        annotations=ToolAnnotations(
            readOnlyHint=True,
            openWorldHint=False,
        ),
        description=(
            "Look up SAP table maintenance view entries from SM30 (read-only display mode). "
            "USE THIS instead of sap_transaction('SM30') - faster and returns structured data. "
            "Returns all entries from the maintenance view with dynamically-parsed columns. "
            "Supports any flat table view (e.g., V_T005, V_T002, custom Z* views). "
            "Non-flat views (SM34/cluster/hierarchical) are detected and reported as unsupported."
        ),
    )
    async def sap_sm30_lookup(
        view_name: str,
        output_file: str | None = None,
        session: str | None = None,
        agent_id: str | None = None,
    ) -> SM30ViewResult | SM30FileSummary:
        """
        Look up table maintenance view entries from SM30.

        Args:
            view_name: The maintenance view or table name (e.g., 'V_T005', 'V_T002')
            output_file: If provided, write full results to this JSON file and return summary.
            session: Session ID (e.g., "s1", "s2"). None uses primary session.
            agent_id: Agent identifier for binding check. Optional.

        Returns:
            SM30ViewResult with view data (inline), or
            SM30FileSummary with file path and preview (when output_file provided)
        """
        now = datetime.now(UTC)
        browser_manager = await get_browser_manager()

        try:
            page = browser_manager.get_session_page_checked(
                session, agent_id, "sap_sm30_lookup"
            )
        except ValueError as e:
            return SM30ViewResult.failure(
                error=f"Session error: {e}",
                view_name=view_name,
                description="",
                view_type="unsupported",
                columns=[],
                rows=[],
                row_count=0,
                retrieved_at=now,
            )

        try:
            result = await _lookup_view(page, view_name)
        except Exception as e:  # pylint: disable=broad-exception-caught
            logger.exception("Looking up SM30 view", extra={"view": view_name})
            result = SM30ViewResult.failure(
                error=f"Error looking up view '{view_name}': {e}",
                view_name=view_name,
                description="",
                view_type="unsupported",
                columns=[],
                rows=[],
                row_count=0,
                retrieved_at=now,
            )

        # Write to file if requested
        if output_file and result.success:
            output_path = Path(output_file)
            output_path.parent.mkdir(parents=True, exist_ok=True)

            with output_path.open("w", encoding="utf-8") as f:
                json.dump(
                    result.model_dump(mode="json"), f, indent=2, ensure_ascii=False
                )

            return SM30FileSummary(
                success=True,
                output_file=str(output_path.absolute()),
                view_name=result.view_name,
                description=result.description,
                view_type=result.view_type,
                columns=result.columns,
                row_count=result.row_count,
                sample_rows=result.rows[:5],
            )

        return result
```
